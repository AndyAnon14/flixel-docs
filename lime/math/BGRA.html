abstract BGRA(Int)

package lime.math

from UInt, Int, to UInt, Int

A utility for storing, accessing and converting colors in a BGRA (blue, green, red, alpha) color format.

var color:BGRA = 0x003388FF;
trace (color.b); // 0x00
trace (color.g); // 0x33
trace (color.r); // 0x88
trace (color.a); // 0xFF

var convert:ARGB = color; // 0xFF883300
